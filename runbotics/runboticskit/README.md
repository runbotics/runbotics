## Install

```bash
npm install --save runbotickit
```

## Usage

```tsx
import React, { Component } from 'react';

import MyComponent from 'runbotickit';
import 'runbotickit/dist/index.css';

class Example extends Component {
    render() {
        return <MyComponent />;
    }
}
```
